int cop(int a[],int b[])
{
    if (a[0] < b[0]) return -1;
    else if (a[0] > b[0]) return 1;
    else
    {
        for (int i=a[0]; i>0; i--)
        {
            if (a[i] > b[i]) return 1;
            else if (a[i] < b[i]) return -1;
        }
        return 0;
    }
}
