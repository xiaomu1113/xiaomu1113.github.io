void quick(int l,int r,int a[])
{
    int i,j,temp,mid;
    i = l;
    j = r;
    mid = a[(i+j)/2];
    do
    {
        while (a[i] < mid) i++;
        while (a[j] > mid) j--;
        if (i <= j)
        {
            temp = a[i];
            a[i] = a[j];
            a[j] = temp;
            i++;
            j--;
        }
    }while (i <= j);
    if (j > l) quick(1,j,dt);
    if (i < r) quick(i,n,dt);
}

void insertion(int a[])
{
    for (int i=1; i<=n; i++)
    {
        int key = a[i];
        int j = i-1;
        while (j>0 && a[j] > key)
        {
            a[j+1] = a[j];
            j--;
        }
        a[j+1] = key;
    }
